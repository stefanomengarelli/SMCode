        /// <summary>Returns SQL expression by sample derived from controls in the form.
        /// Will be considered only controls wich name starting with parameter passed.
        /// Expression will be build according with database type.</summary>
        public static string SqlByControls(Form _Form, SMDatabaseType _DatabaseType, string _ControlNameStartWith)
        {
            if (_Form != null)
            {
                _ControlNameStartWith = _ControlNameStartWith.Trim();
                return SqlByControl((Control)_Form, _DatabaseType, _ControlNameStartWith);
            }
            else return "";
        }

        /// <summary>Returns SQL expression by sample derived from control and its childs.
        /// Will be considered only controls wich name starting with parameter passed.
        /// Expression will be build according with database type.</summary>
        public static string SqlByControl(Control _Control, SMDatabaseType _DatabaseType, string _ControlNameStartWith)
        {
            int i;
            string q;
            StringBuilder r = new StringBuilder();
            if ((_ControlNameStartWith == "") || (_Control.Name.StartsWith(_ControlNameStartWith)))
            {
                if (_Control is SMTextBox) q = ((SMTextBox)_Control).QueryBySample(_DatabaseType).Trim();
                else if (_Control is SMCheckBox) q = ((SMCheckBox)_Control).QueryBySample(_DatabaseType).Trim();
                else if (_Control is SMRadioButton) q = ((SMRadioButton)_Control).QueryBySample(_DatabaseType).Trim();
                else if (_Control is SMButton) q = ((SMButton)_Control).QueryBySample(_DatabaseType).Trim();
                else q = "";
                if (q.Length > 0) 
                {
                    r.Append('(');
                    r.Append(q);
                    r.Append(')');
                }
            }
            for (i = 0; i < _Control.Controls.Count; i++)
            {
                q = SqlByControl(_Control.Controls[i], _DatabaseType, _ControlNameStartWith).Trim();
                if (q.Length > 0)
                {
                    if (r.Length > 0) r.Append("AND");
                    r.Append(q);
                }
            }
            return r.ToString();
        }

        /// <summary>Create _dbinfo table on database dropping eventually existing table. Returns true if succeed.</summary>
        public static bool SqlCreateDbInfo(SMDatabase _Database)
        {
            bool r = false;
            string s;
            SMDataSet ds;
            if (_Database.Type == SMDatabaseType.MySql)
            {
                s = "CREATE TABLE `_dbinfo` ("
                  + "`ID` VARCHAR(12),"
                  + "`dbiTable` VARCHAR(64),"
                  + "`dbiVersion` BIGINT(20),"
                  + "`dbiKeys` VARCHAR(128),"
                  + "`dbiIndexList` VARCHAR(254),"
                  + "`dbiTag` VARCHAR(240),"
                  + "PRIMARY KEY (`ID`),"
                  + "UNIQUE KEY IX_KEY (`dbiTable`))";
            }
            else
            {
                s = "CREATE TABLE [_dbinfo] ("
                  + "[ID] VARCHAR(12) PRIMARY KEY,"
                  + "[dbiTable] VARCHAR(64),"
                  + "[dbiVersion] BIGINT,"
                  + "[dbiKeys] VARCHAR(128),"
                  + "[dbiIndexList] VARCHAR(254),"
                  + "[dbiTag] VARCHAR(240),"
                  + "CONSTRAINT IX_KEY UNIQUE ([dbiTable]))";
            }
            if (_Database.Open())
            {
                ds = new SMDataSet(_Database);
                ds.Exec("DROP TABLE _dbInfo");
                r = ds.Exec(s) > -1;
                ds.Close();
            }
            return r;
        }

        /// <summary>Return string containing SQL parameterized command for dataset structure 
        /// create command with database type syntax.</summary>
        public static string SqlCreateString(SMDataSet _DataSet, SMDatabaseType _DatabaseType)
        {
            int i = 0;
            string s = " ";
            DataColumn c;
            StringBuilder r = new StringBuilder();
            r.Append("CREATE TABLE ");
            r.Append(SM.QuotedField(_DataSet.TableName, _DatabaseType));
            r.Append(" (");
            if (_DatabaseType == SMDatabaseType.None) _DatabaseType = _DataSet.Database.Type;
            if (_DatabaseType == SMDatabaseType.MySql)
            {
                while (i < _DataSet.Table.Columns.Count)
                {
                    c = _DataSet.Table.Columns[i];
                    r.Append(s);
                    r.Append(SM.QuotedField(c.ColumnName, _DatabaseType));
                    r.Append(" ");
                    r.Append(SM.SqlColumnType(c, _DatabaseType));
                    s = ", ";
                    i++;
                }
                if (_DataSet.UniqueID != SMUniqueIDType.None)
                {
                    r.Append(", PRIMARY KEY (");
                    r.Append(SM.QuotedField("ID", _DatabaseType));
                    r.Append(")");
                }
                r.Append(")");
            }
            else
            {
                while (i < _DataSet.Table.Columns.Count)
                {
                    c = _DataSet.Table.Columns[i];
                    r.Append(s);
                    r.Append(SM.QuotedField(c.ColumnName, _DatabaseType));
                    r.Append(" ");
                    r.Append(SM.SqlColumnType(c, _DatabaseType));
                    if ((c.ColumnName == "ID") && (_DataSet.UniqueID != SMUniqueIDType.None)) r.Append(" PRIMARY KEY");
                    s = ", ";
                    i++;
                }
                r.Append(")");
            }
            return r.ToString();
        }

        /// <summary>Return string containing sql statement with [ ] delimiters 
        /// turned in to ty type database delimiters.</summary>
        public static string SqlDelimiters(string _SQLStatement, SMDatabaseType _DatabaseType)
        {
            int i;
            bool b;
            char c, c1, c2;
            StringBuilder r;
            if (_DatabaseType == SMDatabaseType.MySql)
            {
                b = false;
                r = new StringBuilder();
                c1 = SM.databaseMySqlPrefix[0];
                c2 = SM.databaseMySqlSuffix[0];
                for (i = 0; i < _SQLStatement.Length; i++)
                {
                    c = _SQLStatement[i];
                    if (c == '\'') b = !b;
                    else if ((c == SM.databaseSqlPrefix[0]) && !b) c = c1;
                    else if ((c == SM.databaseSqlSuffix[0]) && !b) c = c2;
                    r.Append(c);
                }
                return r.ToString();
            }
            else return _SQLStatement;
        }

        /// <summary>Executes SQL statement passed as parameter. Is statement start by SELECT
        /// function will return integer value of result of first column of first row 
        /// else will return the number of records affected or -1 if not succeed.</summary>
        public static int SqlExec(SMDatabase _Database, string _SqlStatement, bool _ErrorManagement)
        {
            int r = -1;
            bool q = false;
            SM.ErrorTracePush("[Sql.cs] - SM.SqlExec([" + SM.Iif(_Database != null, _Database.Database, "null") + "], " +
                SM.QuotedStr(_SqlStatement) + ", " + SM.Iif(_ErrorManagement, "true", "false") + ")");
            if (_Database != null)
            {
                if (_Database.Keep())
                {
                    _SqlStatement = SM.SqlDelimiters(SM.SqlMacros(_SqlStatement, _Database.Type), _Database.Type).Trim();
                    q = _SqlStatement.ToUpper().StartsWith("SELECT ");
                    try
                    {
                        if (_Database.Type == SMDatabaseType.Access)
                        {
                            OleDbCommand cmd = new OleDbCommand(_SqlStatement, _Database.OleDB);
                            if (q) r = SM.StrToInt(cmd.ExecuteScalar().ToString());
                            else r = cmd.ExecuteNonQuery();
                        }
                        else if (_Database.Type == SMDatabaseType.DBase4)
                        {
                            OleDbCommand cmd = new OleDbCommand(_SqlStatement, _Database.OleDB);
                            if (q) r = SM.StrToInt(cmd.ExecuteScalar().ToString());
                            else r = cmd.ExecuteNonQuery();
                        }
                        else if (_Database.Type == SMDatabaseType.MySql)
                        {
                            MySqlCommand cmd = new MySqlCommand(_SqlStatement, _Database.MySqlDB);
                            if (q) r = SM.StrToInt(cmd.ExecuteScalar().ToString());
                            else r = cmd.ExecuteNonQuery();
                        }
                        else
                        {
                            SqlCommand cmd = new SqlCommand(_SqlStatement, _Database.SqlDB);
                            if (q) r = SM.StrToInt(cmd.ExecuteScalar().ToString());
                            else r = cmd.ExecuteNonQuery();
                        }
                        if (r < 0) r = 0;
                    }
                    catch (Exception ex)
                    {
                        if (_ErrorManagement)
                        {
                            SM.Error(ex);
                            SM.ErrorMessage += SM.CR + SM.CR + "*** SQL STATEMENT ***" + SM.CR + _SqlStatement;
                        }
                        r = -1;
                    }
                }
            }
            SM.ErrorTracePop();
            return r;
        }

        /// <summary>Performs table creation performing all after create scripts.</summary>
        public static bool SqlIndexesCreation(SMDataSet _SourceDataSet, SMDataSet _TargetDataSet)
        {
            int i;
            bool r = false;
            string sql, ixs, ix;
            SMDataSet ds;
            try
            {
                if ((_SourceDataSet != null) && (_TargetDataSet != null))
                {
                    if ((_TargetDataSet.Database.Type != SMDatabaseType.Access)
                        && (_TargetDataSet.Database.Type != SMDatabaseType.DBase4)
                        && (_TargetDataSet.Database.Type != SMDatabaseType.None))
                    {
                        ds = new SMDataSet(_SourceDataSet.Database);
                        if (ds.OpenAtLeast("SELECT * FROM _dbInfo WHERE dbiTable=" + SM.QuotedStr(_SourceDataSet.TableName)))
                        {
                            if (ds.IsField("dbiKeys"))
                            {
                                ixs = ds.FieldStr("dbiKeys").Trim();
                                i = 0;
                                while ((i < 16) && (ixs.Trim().Length > 0))
                                {
                                    ix = SM.StrExtract(ref ixs, "|;").Trim();
                                    if (ix.Length > 1)
                                    {
                                        if ((_TargetDataSet.UniqueID == SMUniqueIDType.None) || ((ix != "ID") && (ix != "!ID") && (ix != "#ID")))
                                        {
                                            i++;
                                            if (ix.StartsWith("#") || ix.StartsWith("!"))
                                            {
                                                sql = "CREATE UNIQUE INDEX ux" + i.ToString() + " ON " + _SourceDataSet.TableName + " (" + ix.Substring(1) + ")";
                                            }
                                            else sql = "CREATE INDEX ix" + i.ToString() + " ON " + _SourceDataSet.TableName + " (" + ix + ")";
                                            if (_TargetDataSet.Exec(sql) > -1) r = true;
                                        }
                                    }
                                }
                            }
                        }
                        ds.Close();
                        ds.Dispose();
                    }
                }
            }
            catch (Exception ex)
            {
                SM.Error(ex);
                r = false;
            }
            return r;
        }

        /// <summary>Return string containing SQL parameterized syntax for dataset insert command.
        /// Require use of unique IDs.</summary>
        public static string SqlInsertCommandString(SMDataSet _DataSet)
        {
            string q = "";
            StringBuilder r = new StringBuilder(), v = new StringBuilder();
            r.Append("INSERT INTO ");
            r.Append(SM.QuotedField(_DataSet.TableName, _DataSet.Database.Type));
            r.Append(" (");
            for (int i = 0; i < _DataSet.Table.Columns.Count; i++)
            {
                if (!_DataSet.Table.Columns[i].AutoIncrement)
                {
                    r.Append(q);
                    r.Append(SM.QuotedField(_DataSet.Table.Columns[i].ColumnName, _DataSet.Database.Type));
                    v.Append(q);
                    v.Append('@');
                    v.Append(_DataSet.Table.Columns[i].ColumnName);
                    q = ",";
                }
            }
            r.Append(") VALUES(");
            r.Append(v.ToString());
            r.Append(')');
            return r.ToString();
        }

        /// <summary>Return string with SQL expression for INSTR of expr, below database type.</summary>
        public static string SqlInstr(string _SQLExpression, SMDatabaseType _DatabaseType)
        {
            if (_SQLExpression.Trim().Length > 0)
            {
                if (_DatabaseType == SMDatabaseType.MySql) return "INSTR(" + _SQLExpression.Trim() + ")";

        /// <summary>Return string replacing SM SQL Macros with corresponding database SQL statement.
        /// Macros: SM_ISNULL(expr), SM_UPPER(expr), SM_LOWER(expr), SM_TRIM(expr), SM_INSTR(expr,substring),
        /// SM_DATETIME(date), SM_USER(), SM_NOW().</summary>
        public static string SqlMacros(string _SQLStatement, SMDatabaseType _DatabaseType)
        {
            string a, m, q;
            //
            // macro SM_ISNULL()
            //
            m = "SM_ISNULL";
            a = SM.StrBtwU(_SQLStatement, m + "(", ")").Trim();
            while (a.Length > 0)
            {
                q = SM.SqlIsNull(a, _DatabaseType);
                _SQLStatement = _SQLStatement.Replace(m + "(" + a + ")", "(" + q + ")");
                a = SM.StrBtw(_SQLStatement, m + "(", ")").Trim();
            }
            //
            // macro SM_UPPER()
            //
            m = "SM_UPPER";
            a = SM.StrBtwU(_SQLStatement, m + "(", ")").Trim();
            while (a.Length > 0)
            {
                q = SM.SqlUpper(a, _DatabaseType);
                _SQLStatement = _SQLStatement.Replace(m + "(" + a + ")", "(" + q + ")");
                a = SM.StrBtw(_SQLStatement, m + "(", ")").Trim();
            }
            //
            // macro SM_LOWER()
            //
            m = "SM_LOWER";
            a = SM.StrBtwU(_SQLStatement, m + "(", ")").Trim();
            while (a.Length > 0)
            {
                q = SM.SqlLower(a, _DatabaseType);
                _SQLStatement = _SQLStatement.Replace(m + "(" + a + ")", "(" + q + ")");
                a = SM.StrBtw(_SQLStatement, m + "(", ")").Trim();
            }
            //
            // macro SM_TRIM()
            //
            m = "SM_TRIM";
            a = SM.StrBtwU(_SQLStatement, m + "(", ")").Trim();
            while (a.Length > 0)
            {
                q = SM.SqlTrim(a, _DatabaseType);
                _SQLStatement = _SQLStatement.Replace(m + "(" + a + ")", "(" + q + ")");
                a = SM.StrBtw(_SQLStatement, m + "(", ")").Trim();
            }
            //
            // macro SM_INSTR()
            //
            m = "SM_INSTR";
            a = SM.StrBtwU(_SQLStatement, m + "(", ")").Trim();
            while (a.Length > 0)
            {
                q = SM.SqlInstr(a, _DatabaseType);
                _SQLStatement = _SQLStatement.Replace(m + "(" + a + ")", "(" + q + ")");
                a = SM.StrBtw(_SQLStatement, m + "(", ")").Trim();
            }
            //
            // macro SM_DATETIME()
            //
            m = "SM_DATETIME";
            a = SM.StrBtwU(_SQLStatement, m + "(", ")").Trim();
            while (a.Length > 0)
            {
                q = SM.QuotedDate(SM.CToDT(a), _DatabaseType);
                _SQLStatement = _SQLStatement.Replace(m + "(" + a + ")", "(" + q + ")");
                a = SM.StrBtw(_SQLStatement, m + "(", ")").Trim();
            }
            //
            // macro SM_SYSUSER()
            //
            m = "SM_USER()";
            while (_SQLStatement.IndexOf(m) > -1)
            {
                _SQLStatement = _SQLStatement.Replace(m, SM.QuotedStr(SM.Users.Current.User));
            }
            //
            // macro SM_NOW()
            //
            m = "SM_NOW()";
            while (_SQLStatement.IndexOf(m) > -1)
            {
                _SQLStatement = _SQLStatement.Replace(m, SM.QuotedDate(DateTime.Now, _DatabaseType));
            }
            //
            // ritorna il risultato
            //
            return _SQLStatement;
        }

        /// <summary>Open the query specified in SQL selection on database and returns the number 
        /// of records loaded or -1 if the function fails. Records are store in tagged 
        /// way in the database clipboard list until max records reached if > -1</summary>
        public static int SqlOpen(SMDatabase _Database, string _SQLSelection, int _MaxRecords)
        {
            int r = -1;
            SMDataSet ds;
            if (_Database != null)
            {
                try
                {
                    ds = new SMDataSet(_Database);
                    dataClipboard.Clear();
                    if (ds.Open(_SQLSelection))
                    {
                        r = 0;
                        while (!ds.Eof && ((r < 0) || (r < _MaxRecords)))
                        {
                            dataClipboard.Add(ds.RecordToString(false));
                            r++;
                            ds.Next();
                        }
                    }
                    ds.Close();
                    ds.Dispose();
                }
                catch (Exception ex)
                {
                    SM.Error(ex);
                    r = -1;
                }
            }
            return r;
        }


        /// <summary>Performs SQL script file on database. If script start with @ all script on Library\Database\Scripts\MySql
        /// or Library\Database\Scripts\Sql depending on database type
        /// directory matching script will be performed. Returns record affected or -1 if fail.</summary>
        public static int SqlPerformScripts(SMDatabase _Database, string _SQLScriptFile)
        {
            int r = -1, i, h, q;
            string pth;
            string[] sql = null;
            List<string> sl;
            SMDataSet ds;
            _SQLScriptFile = _SQLScriptFile.Trim();
            //
            // create sql script array
            //
            if (_SQLScriptFile.Length > 1)
            {
                if (_SQLScriptFile.StartsWith("@"))
                {
                    _SQLScriptFile = _SQLScriptFile.Substring(1).Trim();
                    if (_Database.Type == SMDatabaseType.MySql) pth = SM.LibPath("Scripts","MySql");
                    else pth = SM.LibPath("Scripts","Sql");
                    pth = SM.FixPath(pth, _SQLScriptFile, "");
                    sl = SM.FileList(pth, true);
                    if (sl != null)
                    {
                        if (sl.Count > 0)
                        {
                            sql = new string[sl.Count];
                            for (i = 0; i < sl.Count; i++)
                            {
                                sql[i] = "";
                                pth = sl[i].Trim();
                                if (pth.Length > 0)
                                {
                                    if (SM.FileExists(pth))
                                    {
                                        pth = SM.LoadString(pth).Trim();
                                        if (pth.Length > 0) sql[i] = pth;
                                    }
                                }
                            }
                        }
                    }
                }
                else sql = new string[] { _SQLScriptFile };
            }
            //
            // perform query scripts
            //
            if ((sql != null) && (_Database != null))
            {
                h = sql.Length;
                if (h > 0)
                {
                    try
                    {
                        ds = new SMDataSet(_Database);
                        i = 0;
                        q = 0;
                        r = 0;
                        while ((q > -1) && (i < h))
                        {
                            if (sql[i].Trim().Length > 0)
                            {
                                q = ds.Exec(sql[i]);
                                if (q > -1) r += q;
                                else r = -1;
                            }
                            i++;
                        }
                        ds.Close();
                        ds.Dispose();
                    }
                    catch (Exception ex)
                    {
                        r = -1;
                        SM.Error(ex);
                    }
                }
            }
            else r = 0;
            //
            // end
            //
            return r;
        }

        /// <summary>Returns query condition with field name, query test and value with format. Test can assume following values: 
        /// = , &gt;, &lt; &gt;=, &lt;=, &lt;&gt; and .. or $ for in-string values and ! for boolean value. Test can be preceeded
        /// by # for numeric values and by &amp; for date values or get type from format string.</summary>
        public static string SqlQueryBySample(SMDatabaseType _DatabaseType, string _FieldName, string _Test, string _Value, string _Format)
        {
            int t = 0;
            string r = "";
            //
            // initialize parameters
            // 
            _FieldName = _FieldName.Trim();
            _Test = _Test.Trim().ToLower();
            _Format = _Format.Trim().ToUpper();
            //
            // validate parameters
            // 
            if ((_Value.Trim().Length > 0) && (_FieldName.Length > 0) && (_Test.Length > 0))
            {
                //
                // detect data types
                // 
                if (_Test.StartsWith("#"))
                {
                    t = 1;
                    _Test = SM.Mid(_Test, 1, _Test.Length).Trim();
                }
                else if (_Format.StartsWith("#") || _Format.StartsWith("&%") || _Format.StartsWith("&EURO")) t = 1;
                else if (_Test.StartsWith("&"))
                {
                    t = 2;
                    _Test = SM.Mid(_Test, 1, _Test.Length).Trim();
                }
                else if (_Format.StartsWith("&D") || _Format.StartsWith("&DT")) t = 2;
                //
                // build test
                // 
                r = SM.QuotedField(_FieldName.Trim(), _DatabaseType);
                if ((_Test == "=") || (_Test == ">") || (_Test == "<") || (_Test == ">=") || (_Test == "<=") || (_Test == "<>"))
                {
                    if (t == 1) r += _Test + SM.StrVal(_Value).ToString("##############0.############");
                    else if (t == 2)
                    {
                        if (_Test == "<=") r += "<" + SM.QuotedDate(SM.CToD(_Value).AddTicks(TimeSpan.TicksPerDay), _DatabaseType);
                        else r += _Test + SM.QuotedDate(SM.CToD(_Value), _DatabaseType);
                    }
                    else r += _Test + SM.QuotedStr(_Value);
                }
                else if ((_Test == "..") || (_Test == "$")) r += " LIKE " + SM.QuotedStr("%" + _Value + "%");
                else if (_Test == "!") r += "=" + SM.QuotedStr(SM.BoolStr(SM.BoolStr(_Value.Trim())));
                else r = "";
                //
                // end
                // 
            }
            return r;
        }

        /// <summary>Return result field value of table first record with field greather than value related to database alias.</summary>
        public static string SqlRecordNext(string _Alias, string _TableName, string _FieldName, string _Value, string _ResultField)
        {
            string r = "";
            SMDataSet ds;
            if (_Alias.Trim().Length < 1) _Alias = "MAIN";
            ds = new SMDataSet(_Alias);
            if (ds.Open("SELECT [ID],[" + _FieldName + "],[" + _ResultField + "] FROM [" + _TableName + "]"
                + " WHERE [" + _FieldName + "]>" + SM.QuotedStr(_Value) + " ORDER BY [" + _FieldName + "]"))
            {
                if (!ds.Eof) r = ds.FieldStr(_ResultField);
            }
            ds.Close();
            ds.Dispose();
            return r;
        }

        /// <summary>Return result field value of table first record with field less than value related to database alias.</summary>
        public static string SqlRecordPrior(string _Alias, string _TableName, string _FieldName, string _Value, string _ResultField)

            SMDataSet ds;
            if (_Alias.Trim().Length < 1) _Alias = "MAIN";

            }
            ds.Close();
            ds.Dispose();
            return r;
        }

        /// <summary>Performs table creation performing all @aftercreate_[tablename]* scripts.</summary>
        public static bool SqlTableCreation(SMDataSet _SourceDataSet, SMDataSet _TargetDataSet, bool _DropTarget)
        {
            bool r = false;
            string sql;
            if ((_SourceDataSet != null) && (_TargetDataSet != null))
            {
                sql = SqlCreateString(_SourceDataSet, _TargetDataSet.Database.Type);
                _TargetDataSet.Close();
                if (_DropTarget) _TargetDataSet.Exec("DROP TABLE [" + _SourceDataSet.TableName + "]");
                _TargetDataSet.Close();
                r = _TargetDataSet.Exec(sql) > -1;
                _TargetDataSet.Close();
                SqlIndexesCreation(_SourceDataSet, _TargetDataSet);
                if (r) r = SqlPerformScripts(_TargetDataSet.Database, "@aftercreate_" + _SourceDataSet.TableName + "*.sql") > -1;
            }
            return r;
        }

        /// <summary>Return table name from SQL selection statement.</summary>
        public static string SqlTableName(string _SQLSelection)
        {
            int i = _SQLSelection.ToLower().IndexOf(" from ");
            if (i > -1)
            {
                _SQLSelection = SM.Mid(_SQLSelection, i + 6, 255);
                return SM.Mid(_SQLSelection, 0, SM.ChrPos(" ;,", _SQLSelection + " ") - 1).Trim(new char[] { ' ', '[', ']' });
            }
            else return "";
        }

        /// <summary>Return string with SQL expression for TRIM, below database type.</summary>
        public static string SqlTrim(string _SQLExpression, SMDatabaseType _DatabaseType)
        {
            if (_SQLExpression.Trim().Length > 0)
            {
                if (_DatabaseType == SMDatabaseType.MySql) return "TRIM(" + _SQLExpression.Trim() + ")";
                else if (_DatabaseType == SMDatabaseType.Sql) return "LTRIM(RTRIM(" + _SQLExpression.Trim() + "))";
                else if (_DatabaseType == SMDatabaseType.SqlExpress) return "LTRIM(RTRIM(" + _SQLExpression.Trim() + "))";
                else return "Trim(" + _SQLExpression + ")";
            }
            else return "";
        }

        /// <summary>Return string with SQL expression for UPPERCASE, below database type.</summary>
        public static string SqlUpper(string _SQLExpression, SMDatabaseType _DatabaseType)
        {
            if (_SQLExpression.Trim().Length > 0)
            {
                if (_DatabaseType == SMDatabaseType.MySql) return "UCASE(" + _SQLExpression.Trim() + ")";
                else if (_DatabaseType == SMDatabaseType.Sql) return "UPPER(" + _SQLExpression.Trim() + ")";
                else if (_DatabaseType == SMDatabaseType.SqlExpress) return "UPPER(" + _SQLExpression.Trim() + ")";
                else return "UCase(" + _SQLExpression + ")";
            }
            else return "";
        }

        /// <summary>Return string SQL format for float numeric value.</summary>
        public static string SqlVal(double _Value)
        {
            return _Value.ToString("###############0.############").Replace(",", ".");
        }

        /// <summary>Return string SQL format for float numeric value.</summary>
        public static string SqlVal(string _Value)
        {
            return SqlVal(SM.StrVal(_Value));
        }
