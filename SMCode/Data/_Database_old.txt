        /// <summary>Copy records corresponding to sql selection from source database 
        /// to mdb template database copied on target file or print merge directory if omitted.</summary>
        public static int CopyRecords(SMDatabase _SourceDatabase, string _SQLSelection, bool _ResetTarget, 
            string _MdbFileTarget, SMOnDataRecordCopy _OnDataRecordCopy)
        {
            int r = -1;
            string fileTemplate;
            SMDatabase dbTarget;
            if (_SourceDatabase != null)
            {
                try
                {
                    fileTemplate = _SourceDatabase.TemplatePath();
                    if (_MdbFileTarget.Trim().Length<1) _MdbFileTarget = SM.PrintMergePath(fileTemplate);
                    if (_ResetTarget)
                    {
                        SM.FileDelete(_MdbFileTarget, SM.ioRetryTimes);
                        _ResetTarget = SM.FileCopy(fileTemplate, _MdbFileTarget);
                    }
                    else if (SM.FileExists(_MdbFileTarget)) _ResetTarget = true;
                    else _ResetTarget = SM.FileCopy(fileTemplate, _MdbFileTarget);
                    if (_ResetTarget)
                    {
                        dbTarget = new SMDatabase();
                        if (dbTarget.OpenMdb(_MdbFileTarget))
                        {
                            r = CopyRecords(_SourceDatabase, _SQLSelection, dbTarget, "", true, null, _OnDataRecordCopy);
                        }
                        dbTarget.Close();
                        dbTarget.Dispose();
                    }
                }
                catch (Exception ex)
                {
                    SM.Error(ex);
                }
            }
            return r;
        }

        /// <summary>Copy from source select query of source db to target table of target db all records.
        /// If reset target table is true, target table will be deleted before adding records.
        /// If fields not null will be assigned values by fields array ( "fieldTarget1", 
        /// "fieldSource1", ... "fieldTargetN", "fieldSourceN" ).</summary>
        public static int CopyRecords(SMDatabase _SourceDatabase, string _SQLSelection, SMDatabase _TargetDatabase, 
            string _TargetTableName, bool _ResetTarget, string[] _FieldNames, SMOnDataRecordCopy _OnDataRecordCopy)
        {
            int r = -1, recordIndex, recordTotal;
            bool validate;
            SMDataSet dsSource, dsTarget;
            if (_TargetDatabase == null) _TargetDatabase = _SourceDatabase;
            if (_SourceDatabase != null)
            {
                try
                {
                    dsSource = new SMDataSet(_SourceDatabase);
                    dsTarget = new SMDataSet(_TargetDatabase);
                    if (dsSource.Open(_SQLSelection, true))
                    {
                        if (_TargetTableName.Trim().Length < 1) _TargetTableName = dsSource.TableName;
                        if (_ResetTarget) SM.SqlExec(_TargetDatabase, "DELETE FROM " + _TargetTableName, true);
                        if (dsTarget.Open("SELECT * FROM " + _TargetTableName))
                        {
                            r = 0;
                            recordIndex = 0;
                            recordTotal = dsTarget.RecordCount();
                            while (!SM.StopNow && !dsSource.Eof && (r > -1))
                            {
                                if (_FieldNames != null)
                                {
                                    if (dsTarget.Append())
                                    {
                                        if (dsTarget.Assign(dsSource, _FieldNames))
                                        {
                                            validate = true;
                                            if (_OnDataRecordCopy != null) _OnDataRecordCopy(dsSource, dsTarget, recordIndex, recordTotal, ref validate);
                                            if (validate)
                                            {
                                                if (dsTarget.Post())
                                                {
                                                    r++;
                                                    recordIndex++;
                                                }
                                                else
                                                {
                                                    dsTarget.Cancel();
                                                    r = -1;
                                                }
                                            }
                                            else dsTarget.Cancel();
                                        }
                                        else
                                        {
                                            dsTarget.Cancel();
                                            r = -1;
                                        }
                                    }
                                    else r = -1;
                                }
                                else if (dsTarget.Append())
                                {
                                    if (dsTarget.CopyRow(dsSource))
                                    {
                                        validate = true;
                                        if (_OnDataRecordCopy != null) _OnDataRecordCopy(dsSource, dsTarget, recordIndex, recordTotal, ref validate);
                                        if (validate)
                                        {
                                            if (dsTarget.Post())
                                            {
                                                r++;
                                                recordIndex++;
                                            }
                                            else
                                            {
                                                dsTarget.Cancel();
                                                r = -1;
                                            }
                                        }
                                        else dsTarget.Cancel();
                                    }
                                    else
                                    {
                                        dsTarget.Cancel();
                                        r = -1;
                                    }
                                }
                                else r = -1;
                                SM.ProcessEvents();
                                dsSource.Next();
                            }
                        }
                    }
                }
                catch (Exception ex)
                {
                    SM.Error(ex);
                    r = -1;
                }
            }
            return r;
        }

        /// <summary>Compare version between databases in aliases array with their local mdb templates.
        /// Returns true if versions are aligned or false if not.</summary>
        public static bool CheckDatabaseVersion(string[] _Aliases, bool _FixDatabase)
        {
            bool r = true;
            int i;
            string va, vb;
            SMDatabase da, db;
            SM.Error("");
            SM.Databases.CloseAll();
            if (_Aliases == null) _Aliases = SM.Databases.AliasArray();
            if (_Aliases != null)
            {
                try
                {
                    db = new SMDatabase();
                    i = 0;
                    while (r && (i < _Aliases.Length))
                    {
                        da = SM.Databases.Keep(_Aliases[i]);
                        if (da != null)
                        {
                            if (db.OpenMdb(da.TemplatePath()))
                            {
                                va = da.LoadSchema();
                                vb = db.LoadSchema();
                                db.Close();
                                SM.Databases.Close(_Aliases[i]);
                                if (va.Trim().Length < 1)
                                {
                                    SM.ErrorDialog(SM.T("@SM|Non č possibile accedere al database. Il programma terminerŕ."));
                                    r = false;
                                }
                                else if (va.CompareTo(vb) > 0)
                                {
                                    SM.ErrorDialog(SM.T("@SM|%%0%% %%1%% - Controllo versione database", SM.Title, SM.Version, ""),
                                        SM.T("@SM|Il database richiede una versione piů recente del software. Il programma terminerŕ."),
                                        SM.T("@SM|Versione database prevista: " + vb + "\r\nVersione database rilevata: " + va),
                                        "", null);
                                    r = false;
                                }
                                else if (va.CompareTo(vb) < 0)
                                {
                                    if (!_FixDatabase || SM.databaseClientMode)
                                    {
                                        SM.ErrorDialog(SM.T("@SM|%%0%% %%1%% - Controllo versione database", SM.Title, SM.Version, ""),
                                            SM.T("@SM|Il software richiede l'aggiornamento del database alla versione corrente. Il programma terminerŕ."),
                                            SM.T("@SM|Versione database prevista: " + vb + "\r\nVersione database rilevata: " + va),
                                            "", null);
                                        r = false;
                                    }
                                    else
                                    {
                                        r = SM.DataMaintenanceDialog(SM.T("@SM|%%0%% %%1%% - Aggiornamento database", SM.Title, SM.Version, ""), 
                                            "", new string[1] { _Aliases[i] }, false, SM.ThemeImage(@"icons/ui_icon_48_setup.png")) > -1;
                                        if (!r) SM.ErrorDialog(SM.T("@SM|Si č verificato un errore durante la conversione del database. Il programma terminerŕ."), SM.errorException);
                                        else if (!SM.DatabaseMaintenancePerformed)
                                        {
                                            SM.ErrorDialog(SM.T("@SM|%%0%% %%1%% - Aggiornamento database", SM.Title, SM.Version, ""),
                                                SM.T("@SM|La conversione del database č stata annullata. Il programma terminerŕ."),
                                                SM.T("@SM|Versione database prevista: " + vb + "\r\nVersione database rilevata: " + va),
                                                "", null);
                                            r = false;
                                        }
                                    }
                                }
                            }
                            else
                            {
                                SM.ErrorDialog(SM.T("@SM|Si č verificato un errore durante la connessione al template del database. Il programma terminerŕ."), SM.errorException);
                                r = false;
                            }
                        }
                        else
                        {
                            SM.ErrorDialog(SM.T("@SM|Si č verificato un errore durante la connessione al database da comparare. Il programma terminerŕ."), SM.errorException);
                            r = false;
                        }
                        i++;
                    }
                    db.Dispose();
                }
                catch (Exception ex)
                {
                    SM.Error(ex);
                    SM.ErrorDialog(SM.T("@SM|Si č verificato un errore durante la comparazione del database. Il programma terminerŕ."), SM.errorException);
                    r = false;
                }
            }
            return r;
        }

        /// <summary>Compact MDB database file specified in file name parameter. Password must be setted 
        /// to security password to access database file or "" if not necessary. Return true if succeed.</summary>
        public static bool CompactMDB(string _FileName, string _Password)
        {
            bool r = false;
            object jro;
            object[] par;
            string tmp = SM.FixPath(SM.ExtractFilePath(_FileName), SM.ExtractFileNoExt(_FileName) + "_tmp", "mdb");
            string bkp = SM.FixPath(SM.ExtractFilePath(_FileName), SM.ExtractFileNoExt(_FileName) + "_bkp", "mdb");
            string src = "Provider = Microsoft.Jet.OLEDB.4.0; Data Source = " + _FileName;
            string tgt = "Provider = Microsoft.Jet.OLEDB.4.0; Data Source = " + tmp + "; Jet OLEDB:Engine Type=5";
            if (!SM.Empty(_Password)) 
            {
                tgt += "; Jet OLEDB:Database Password = " + _Password;
                src += "; Jet OLEDB:Database Password = " + _Password;
            }
            try
            {
                SM.FileDelete(tmp, SM.IORetryTimes);
                jro = Activator.CreateInstance(Type.GetTypeFromProgID("JRO.JetEngine"));
                par = new object[] { src, tgt };
                jro.GetType().InvokeMember("CompactDatabase", System.Reflection.BindingFlags.InvokeMethod, null, jro, par);
                SM.FileDelete(bkp, SM.IORetryTimes);
                if (SM.FileMove(_FileName, bkp, SM.IORetryTimes)) r = SM.FileMove(tmp, _FileName, SM.IORetryTimes);
                System.Runtime.InteropServices.Marshal.ReleaseComObject(jro);
                jro = null;
            }
            catch (Exception ex)
            {
                SM.Error(ex);
                r = false;
            }
            return r;
        }

        /// <summary>If not exists create connection parameters to target or, if empty, 
        /// to default application INI file with database alias. Returns true if succeed.</summary>
        public static bool CreateAlias(string _TargetIniFilePath, string _Alias, SMDatabaseType _DatabaseType, 
            string _DatabaseHost, string _DatabaseName, string _Template, string _Path, string _User, string _Password)
        {
            string s;
            SMIniFile ini;
            //
            // init values
            //
            _Alias = _Alias.Trim().ToUpper();
            _DatabaseHost = _DatabaseHost.Trim();
            _DatabaseName = _DatabaseName.Trim();
            _Template = _Template.Trim();
            if ((_Template.Length < 1) && (_Alias == "MAIN")) _Template = SM.ExecName;
            _Path = _Path.Trim();
            _User = _User.Trim();
            if (_DatabaseType == SMDatabaseType.Access)
            {
                if (_DatabaseHost.Length < 1) _DatabaseHost = "localhost";
                if (_DatabaseName.Length < 1) _DatabaseName = SM.ExecName;
                if (_Path.Length < 1) _Path = SM.DataDir;
            }
            else if (_DatabaseType == SMDatabaseType.DBase4)
            {
                if (_DatabaseHost.Length < 1) _DatabaseHost = "localhost";
                if (_DatabaseName.Length < 1) _DatabaseName = SM.ExecName;
                if (_Path.Length < 1) _Path = SM.DataDir;
            }
            //
            // validate alias
            //
            if ((_Alias.Length > 0) && (SM.Language != SMLanguage.Auto))
            {
                ini = new SMIniFile(_TargetIniFilePath);
                s = "DATABASE " + _Alias;
                //
                // create type parameter
                //
                if ((SMDatabaseType)ini.ReadInteger(s, "TYPE", (int)SMDatabaseType.None) == SMDatabaseType.None)
                {
                    ini.WriteInteger(s, "TYPE", (int)_DatabaseType);
                }
                //
                // create other database parameters
                //
                ini.CreateString(s, "HOST", _DatabaseHost, true);
                ini.CreateString(s, "NAME", _DatabaseName, true);
                ini.CreateString(s, "TEMPLATE", _Template, true);
                ini.CreateString(s, "PATH", _Path, true);
                ini.CreateString(s, "USER", _User, false);
                ini.CreateString(s, "PASS", _Password, false);
                //
                // save ini
                //
                return ini.Save();
            }
            else return false;
        }

                else if (_Database.Type == SMDatabaseType.DBase4) r = r + SM.FixPath("", _Database.Database, "dbf") + "]";

        /// <summary>Compare column type values from objects. Return more than 0 if first greater 
        /// then second, less than zero if first lesser then second, zero if values are equal.</summary>
        public static int FieldCompare(DataColumn _DataColumn, object _Value1, object _Value2, bool _CaseSensitive)
        {
            if (SM.FieldIsDate(_DataColumn))
            {
                DateTime d1 = CToDT(_Value1.ToString()), d2 = CToDT(_Value2.ToString());
                if (d1 > d2) return 1;
                else if (d1 < d2) return -1;
                else return 0;
            }
            else if (SM.FieldIsNumeric(_DataColumn))
            {
                Double n1 = StrVal(_Value1.ToString()), n2 = StrVal(_Value2.ToString());
                if (n1 > n2) return 1;
                else if (n1 < n2) return -1;
                else return 0;
            }
            else return String.Compare(_Value1.ToString(), _Value2.ToString(), !_CaseSensitive);
        }

        /// <summary>Return true if column field data type is blob.</summary>
        public static string FieldInc(SMDatabase _Database, string _TableName, string _FieldName, string _Prefix, int _Digits)
        {
            string r = "", fld, sql;
            SMDataSet ds;
            if (_Database != null)
            {
                try
                {
                    ds = new SMDataSet(_Database);
                    fld = SM.QuotedField(_FieldName);
                    sql = "SELECT " + fld + " FROM " + SM.QuotedField(_TableName);
                    sql += " WHERE (" + fld + " LIKE " + SM.QuotedStr(_Prefix.Trim() + SM.Replicate("[0-9]", _Digits)) + ")";
                    sql += " ORDER BY " + fld + " DESC";
                    if (ds.Open(sql, true))
                    {
                        if (ds.Eof) r = _Prefix + SM.Zeroes(1, _Digits);
                        else
                        {
                            r = ds.FieldStr(_FieldName);
                            r = _Prefix + SM.Zeroes(SM.StrToInt(SM.Mid(r, _Prefix.Length, r.Length)) + 1, _Digits);
                        }
                    }
                    ds.Close();
                    ds.Dispose();
                }
                catch (Exception ex)
                {
                    SM.Error(ex);
                    r = "";
                }
            }
            return r;
        }

        /// <summary>Load database parameters related to database alias from source or if empty from 
        /// default application INI file. Returns true if succeed.</summary>
        public static bool LoadDatabaseAlias(string _SourceIniFilePath, string _Alias, ref SMDatabaseType _DatabaseType, ref string _DatabaseHost,
            ref string _DatabaseName, ref string _Template, ref string _Path, ref string _User, ref string _Password)
        {
            string s;
            SMIniFile ini;
            _Alias = _Alias.Trim().ToUpper();
            if ((_Alias.Length > 0) && (SM.Language != SMLanguage.Auto))
            {
                ini = new SMIniFile(_SourceIniFilePath);
                s = "DATABASE " + _Alias;
                _DatabaseType = (SMDatabaseType)ini.ReadInteger(s, "TYPE", (int)SMDatabaseType.Access);
                _DatabaseHost = ini.ReadString(s, "HOST", "localhost");
                _DatabaseName = ini.ReadString(s, "NAME", SM.ExecName);
                _Template = ini.ReadString(s, "TEMPLATE", _Template);
                _Path = ini.ReadString(s, "PATH", SM.DataDir);
                _User = ini.ReadString(s, "USER", "");
                _Password = ini.ReadMasked(s, "PASS", SM.DatabaseAccessPassword);
                return true;
            }
            else return false;
        }

        /// <summary>Load parameters from alias and assign to database. If specified, database alias name 
        /// will be assigned otherwise cleared.</summary>
        public static bool LoadDatabaseParameters(SMDatabase _Database, string _Alias, bool _AssignAliasName)
        {
            int i;
            bool r = false;
            if ((_Database != null) && (_Alias != null))
            {
                _Database.Alias = "";
                i = SM.Databases.Find(_Alias);
                if (i > -1)
                {
                    if (SM.Databases.Items[i].Active)
                    {
                        _Database.Type = SM.Databases.Items[i].Type;
                        _Database.Host = SM.Databases.Items[i].Host;
                        _Database.Database = SM.Databases.Items[i].Database;
                        _Database.Template = SM.Databases.Items[i].Template;
                        _Database.Path = SM.Databases.Items[i].Path;
                        _Database.User = SM.Databases.Items[i].User;
                        _Database.Password = SM.Databases.Items[i].Password;
                        r = true;
                    }
                }
                if (!r) r = _Database.Load("", _Alias, false);
                if (r && _AssignAliasName) _Database.Alias = _Alias;
            }
            return r;
        }

        /// <summary>Try to lock database area.</summary>
        public static bool LockDatabaseArea()
        {
            if (!LockedDatabaseArea())
            {
                return SM.SaveString(LockDatabasePath(), 
                    SM.execName + ";" + SM.DTToC(DateTime.Now) 
                    + ";" + SM.CurrentMachine() + ";" + SM.CurrentOSUser());
            }
            else return false;
        }

        /// <summary>Return true if database area is locked.</summary>
        public static bool LockedDatabaseArea()
        {
            string f = LockDatabasePath();
            if (SM.FileExists(f)) return SM.FileDate(f) > DateTime.Now.AddHours(-4);
            else return false;
        }

        /// <summary>Change database table records deep level path from old path to new path.
        /// Return true if succeed.</summary>
        public static bool MoveRecordsPath(SMDatabase _Database, string _TableName, string _OldPath, string _NewPath)
        {
            bool r = true;
            int level;
            string s, parent;
            SMDataSet ds;
            _OldPath = SM.DeepLevel(_OldPath, 999);
            level = SM.DeepLevel(_OldPath);
            parent = SM.DeepLevel(_OldPath, level - 1);
            _NewPath = SM.DeepLevel(_NewPath, 999);
            if ((_OldPath != _NewPath) && (parent != _NewPath))
            {
                ds = new SMDataSet(_Database);
                if (ds.Open("SELECT [ID],[sysPath] FROM [" + _TableName + "] WHERE ([sysPath]="
                    + SM.QuotedStr(_OldPath) + ")OR([sysPath] LIKE " + SM.QuotedStr(_OldPath + "%") + ") ORDER BY [sysPath]"))
                {
                    while (r && !ds.Eof)
                    {
                        s = SM.DeepLevel(ds.Row["sysPath"].ToString(), 999);
                        r = ds.Edit();
                        ds.Row["sysPath"] = SM.DeepLevel(_NewPath + SM.Mid(s, 0, parent.Length), 999);
                        if (r)
                        {
                            r = ds.Post();
                            if (!r) ds.Cancel();
                        }
                        ds.Next();
                    }
                }
                else r = false;
                ds.Close();
                ds.Dispose();
            }
            return r;
        }

        /// <summary>Return dateValue with dbType database SQL syntax delimiters.</summary>
        static public string QuotedDate(DateTime _Value, SMDatabaseType _DatabaseType)
        {
            if ((_DatabaseType == SMDatabaseType.Sql) || (_DatabaseType == SMDatabaseType.SqlExpress))
            {
                // ISO 8601 
                return SM.QuotedStr(Zeroes(_Value.Year, 4) + "-" + Zeroes(_Value.Month, 2) + "-" + Zeroes(_Value.Day, 2)
                       + "T" + Zeroes(_Value.Hour, 2) + ":" + Zeroes(_Value.Minute, 2) + ":" + Zeroes(_Value.Second, 2));
            }
            else if (_DatabaseType == SMDatabaseType.MySql)
            {
                return SM.QuotedStr(Zeroes(_Value.Year, 4) + "-" + Zeroes(_Value.Month, 2) + "-" + Zeroes(_Value.Day, 2)
                       + " " + Zeroes(_Value.Hour, 2) + ":" + Zeroes(_Value.Minute, 2) + ":" + Zeroes(_Value.Second, 2));
            }
            else return "#" + Zeroes(_Value.Month, 2) + "-" + Zeroes(_Value.Day, 2) + "-" + Zeroes(_Value.Year, 4)
                       + " " + Zeroes(_Value.Hour, 2) + "." + Zeroes(_Value.Minute, 2) + "." + Zeroes(_Value.Second, 2) + "#";
        }

        /// <summary>Return field name with standard database SQL syntax [] delimiters.</summary>
        public static string QuotedField(string _FieldName)
        {
            _FieldName = SM.UnquotedField(_FieldName);
            if (_FieldName.Length > 0) return "[" + _FieldName + "]";
            else return "";
        }

        /// <summary>Return field name with type database SQL syntax delimiters.</summary>
        public static string QuotedField(string _FieldName, SMDatabaseType _DatabaseType)
        {
            _FieldName = SM.UnquotedField(_FieldName);
            if ((_DatabaseType == SMDatabaseType.Access)
                || (_DatabaseType == SMDatabaseType.Sql)
                || (_DatabaseType == SMDatabaseType.SqlExpress)) return SM.databaseSqlPrefix + _FieldName + SM.databaseSqlSuffix;
            else if (_DatabaseType == SMDatabaseType.MySql) return SM.databaseMySqlPrefix + _FieldName + SM.databaseMySqlSuffix;
            else return _FieldName;
        }

        /// <summary>Return fields list with type database SQL syntax delimiters.
        /// Fields must be separates by commas.</summary>
        public static string QuotedFieldList(string _FieldsList, SMDatabaseType _DatabaseType)
        {
            return QuotedFieldList(_FieldsList, _DatabaseType, false);
        }

        /// <summary>Return fields list with type database SQL syntax delimiters, with DESC clausole if specified.
        /// Fields must be separates by commas.</summary>
        public static string QuotedFieldList(string _FieldsList, SMDatabaseType _DatabaseType, bool _Descend)
        {
            string c, s = _FieldsList.Trim(), r = SM.StrExtract(ref s, ",;").Trim(), x = SM.Iif(_Descend, " DESC", "");
            if (r.Length > 0) r = QuotedField(r, _DatabaseType) + x;
            while (s.Length > 0)
            {
                c = SM.StrExtract(ref s, ",;").Trim();
                if (c.Length > 0) r += ", " + QuotedField(c, _DatabaseType) + x;
            }
            return r;
        }

        /// <summary>Save database connection parameters to to target or, if empty, default application 
        /// INI file with alias database alias. Returns true if succeed.</summary>
        public static bool SaveAlias(string _TargetIniFilePath, string _Alias, SMDatabaseType _DatabaseType, 
            string _DatabaseHost, string _DatabaseName, string _Template, string _Path, string _User, string _Password)
        {
            bool ok = false;
            string s;
            SMIniFile ini;
            SM.Error("");
            _Alias = _Alias.Trim().ToUpper();
            _DatabaseHost = _DatabaseHost.Trim();
            _DatabaseName = _DatabaseName.Trim();
            _Template = _Template.Trim();
            if (_Template.Length < 1) _Template = SM.ExecName;
            _Path = _Path.Trim();
            _User = _User.Trim();
            if ((SM.Language != SMLanguage.Auto) && (_Alias.Length > 0))
            {
                ini = new SMIniFile(_TargetIniFilePath);
                s = "DATABASE " + _Alias;
                ok = true;
                if (_DatabaseType == SMDatabaseType.Access)
                {
                    if (_DatabaseHost.Length < 1) _DatabaseHost = "localhost";
                    if (_DatabaseName.Length < 1) _DatabaseName = SM.ExecName;
                    if (_Path.Length < 1) _Path = SM.DataDir;
                }
                else if (_DatabaseType == SMDatabaseType.DBase4)
                {
                    if (_DatabaseHost.Length < 1) _DatabaseHost = "localhost";
                    if (_DatabaseName.Length < 1) _DatabaseName = SM.ExecName;
                    if (_Path.Length < 1) _Path = SM.DataDir;
                }
                ini.WriteInteger(s, "TYPE", (int)_DatabaseType);
                ini.WriteString(s, "HOST", _DatabaseHost);
                ini.WriteString(s, "NAME", _DatabaseName);
                ini.WriteString(s, "TEMPLATE", _Template);
                ini.WriteString(s, "PATH", _Path);
                ini.WriteString(s, "USER", _User);
                ini.WriteMasked(s, "PASS", _Password);
                ok = ini.Save();
                if (!ok)
                {
                    SM.Error(SM.T("@SM|Error durante la scrittura della configurazione su:")
                        + " " + SM.ApplicationIniFile(), SM.errorException);
                }
            }
            return ok;
        }

        public static bool UnlockDatabaseArea()
        {
            SM.FileDelete(LockDatabasePath(), SM.ioRetryTimes);
            return !LockedDatabaseArea();
        }

        /// <summary>Return field name without database SQL syntax delimiters.</summary>
        public static string UnquotedField(string _FieldName)
        {
            _FieldName = _FieldName.Trim();
            if (_FieldName.StartsWith("[")
                || _FieldName.StartsWith(SM.databaseSqlPrefix)
                || _FieldName.StartsWith(SM.databaseMySqlPrefix))
            {
                if (_FieldName.Length > 1) _FieldName = _FieldName.Substring(1);
                else _FieldName = "";
            }
            if (_FieldName.EndsWith("]")
                || _FieldName.EndsWith(SM.databaseSqlSuffix)
                || _FieldName.EndsWith(SM.databaseMySqlSuffix))
            {
                if (_FieldName.Length > 1) _FieldName = _FieldName.Substring(0, _FieldName.Length - 1);
                else _FieldName = "";
            }
            return _FieldName.Trim();
        }
